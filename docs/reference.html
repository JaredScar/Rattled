Language Reference

Complete reference for the Rattled programming language — syntax, semantics, and standard library.

Installation

Install Rattled from PyPI. Requires Python 3.8 or newer.

shell terminal
pip install rattled

Or install from source (also sets up the rattled CLI command):

shell terminal
git clone https://github.com/your-username/Rattled
cd Rattled
pip install -e .
# Windows one-click installer:
install.bat
💡

The VS Code extension in vscode-rattled/ provides syntax highlighting for .ry files. Install it by copying to your VS Code extensions folder or using Extensions: Install from VSIX.

Quick Start

Create a file named hello.ry and run it:

.ry hello.ry
` Your first Rattled program `
name = "World"
pr "Hello, {name}!"

fn greet(person: str, times: int = 1) {
    for i in 0..times {
        pr "Hey {person}!"
    }
}

greet("Rattled", 3)
shell run it
rattled hello.ry

Variables & Types

Variables are dynamically typed (like Python). Assign with =. No declaration keyword required.

.ry variables
name    = "Rattled"
version = 7
pi      = 3.14159
active  = TRUE
nothing = null

` Type casting functions `
n = str(42)       ` "42"  `
f = flo("3.14")   ` 3.14  `
i = int(3.99)     ` 3     `
b = boo(0)        ` False `

Global Variables

Use glo to declare a variable in the global scope from inside a function:

.ry
count = 0

fn increment() {
    glo count
    count += 1
}

Strings & Interpolation

Strings use double or single quotes. Embed expressions with {...}:

.ry
name  = "Alice"
score = 95

` Simple variable `
pr "Name: {name}"

` Expression `
pr "Score: {score}"
pr "Pass? {score >= 60}"

` Function call `
pr "Upper: {name.upper()}"
pr "Add: {score + 5}"

` Multi-line string `
msg = """
Hello {name},
Your score is {score}.
"""
pr msg

` pr auto-casts numbers to string `
pr "Value = " + score    ` no str() needed `

Control Flow

if / elif / el

.ry
if score >= 90 {
    pr "A"
} elif score >= 80 {
    pr "B"
} elif score >= 70 {
    pr "C"
} el {
    pr "F"
}

for — three forms

.ry
` 1. Range (inclusive on both ends) `
for i in 0..4 {
    pr i    ` 0 1 2 3 4 `
}

` 2. Collection iteration `
fruits = ["apple", "banana", "cherry"]
for fruit in fruits {
    pr fruit
}

` 3. Dict key-value iteration `
hashm ages = {alice: 30, bob: 25}
for name, age in ages {
    pr "{name} is {age}"
}

` 4. Classic condition-based loop `
for i = 0; i < 5; i++ {
    pr i
}

while

.ry
i = 0
while i < 10 {
    if i % 2 == 0 { cont }    ` skip evens `
    pr i
    i++
    if i == 7 { brk }         ` stop at 7 `
}

Use brk to break out of a loop and cont to skip to the next iteration.

Switch / Case

The sw/cs statement supports equality checks, value guards, and type checks. Use def for a default/fallback branch.

.ry
` Value equality `
sw day {
    cs "Mon" { pr "Start of week" }
    cs "Fri" { pr "End of week"   }
    def      { pr "Midweek"       }
}

` Guard condition (if) `
sw score {
    cs 100          { pr "Perfect!" }
    cs int if score >= 90 { pr "A grade" }
    cs int if score >= 70 { pr "Pass"    }
    def             { pr "Fail"     }
}

` Type-check case `
fn describe(val) {
    sw val {
        cs str  { pr "string: {val}"    }
        cs int  { pr "integer: {val}"   }
        cs list { pr "list: {len(val)} items" }
        def     { pr "unknown"          }
    }
}
💡

Type-check cases use isinstance() under the hood. Supported built-in types: str, int, float, bool, list, dict, tuple, set. Capitalised identifiers (e.g. cs MyClass) also trigger an isinstance check.

Functions

.ry
` Basic function `
fn greet(name) {
    pr "Hello, {name}!"
}

` Default parameter `
fn power(base, exp = 2) {
    ret base ** exp
}

` Variadic positional args `
fn sum(...nums) {
    total = 0
    for n in nums { total += n }
    ret total
}

` Variadic keyword args `
fn config(~~opts) {
    pr opts
}

config(debug = TRUE, timeout = 30)

` Multiple return values `
fn minMax(nums) {
    ret min(nums), max(nums)
}

lo, hi = minMax([3, 1, 7, 2, 9])

Lambdas & Closures

Rattled supports two lambda syntaxes:

.ry
` lam — single expression lambda `
double = lam x -> x * 2
add    = lam x, y -> x + y

` Anonymous fn — multi-line body `
square = fn(n) { ret n * n }

` Used in higher-order contexts `
nums = [5, 3, 8, 1, 9, 2]
nums.sort(key = lam x -> -x)    ` descending `

pairs = [("alice", 95), ("bob", 82)]
pairs.sort(key = lam p -> p[1])  ` sort by score `

` Comprehension with lambda transform `
results = [lam x -> x * x applied for x in 1..5]
` equivalent: `
results = [x * x for x in 1..5]

Type Hints Phase 7

Type annotations are optional and pass through as Python type hints. They have no runtime effect but enable static analysis tools like mypy.

.ry
` Parameter types `
fn greet(name: str, count: int) {
    for i in 0..count { pr "Hi {name}" }
}

` Return type `
fn add(a: int, b: int) -> int {
    ret a + b
}

` Variable annotation `
score: int = 95
name:  str = "Alice"

` Generic types (List[int], Optional[str]) `
fn first(items: list) -> str {
    ret items[0]
}

Arrays

Declare with arr or plain assignment. Arrays are Python lists under the hood.

.ry
arr nums = [1, 2, 3, 4, 5]

` Access and mutate `
first = nums[0]
nums[0] = 10
nums.push(6)
nums.pop()

` Slicing `
slice = nums[1..3]    ` [2, 3]        `
tail  = nums[2..4]    ` [3, 4]        `

` Length `
n = len(nums)

` Built-in sorts `
mergSor(nums)
quikSor(nums)
bubSor(nums)
heapSor(nums)

Hashmaps

Declare with hashm or plain assignment. Hashmaps are Python dicts.

.ry
hashm person = {name: "Alice", age: 30, active: TRUE}

` Access `
pr person["name"]
pr person["age"]

` Update `
person["age"] = 31

` Iterate `
for key, value in person {
    pr "{key}: {value}"
}

Destructuring Assignment Phase 7

.ry
` Basic array destructuring `
[a, b, c] = [1, 2, 3]
pr "a={a} b={b} c={c}"

` Rest / spread in destructuring `
[head, ...tail] = [10, 20, 30, 40, 50]
pr "head={head}  tail={tail}"

` Tuple-style (no brackets) `
x, y = 100, 200
pr "x={x}  y={y}"

` Unpack function return values `
fn rgb(hex) { ret 255, 128, 0 }
r, g, b = rgb("#FF8000")

` Swap two variables `
a, b = b, a

Comprehensions Phase 7

.ry
` List comprehension `
squares = [x * x for x in 1..10]
evens   = [x for x in 1..20 if x % 2 == 0]
upper   = [w.upper() for w in ["hello", "world"]]

` Dict comprehension (single var) `
words   = ["apple", "banana", "cherry"]
lengths = {w: len(w) for w in words}

` Dict comprehension (key-value — iterates .items()) `
hashm prices = {apple: 1, banana: 2, cherry: 3}
doubled  = {k: v * 2 for k, v in prices}
cheap    = {k: v for k, v in prices if v < 3}

pr "lengths: {lengths}"
pr "doubled: {doubled}"
pr "cheap:   {cheap}"

Spread Operator

.ry
` Merge arrays `
a = [1, 2, 3]
b = [4, 5, 6]
c = [...a, ...b]        ` [1,2,3,4,5,6] `

` Pass array as positional args `
fn add3(x, y, z) { ret x + y + z }
nums = [1, 2, 3]
result = add3(...nums)

` In destructuring `
[first, ...rest] = [10, 20, 30, 40]
pr "first={first}  rest={rest}"

Classes

.ry
Clas Animal {
    ` Class-level (static) variable `
    stat kingdom = "Animalia"

    ` Constructor `
    def(name: str, sound: str) {
        self.name  = name
        self.sound = sound
    }

    ` Instance method `
    fn speak() {
        pr "{self.name} says {self.sound}"
    }

    ` Static method `
    stat fn classify() {
        pr "Kingdom: {Animal.kingdom}"
    }
}

dog = Animal("Rex", "Woof")
dog.speak()
Animal.classify()

Inheritance

.ry
Clas Animal {
    def(name: str) { self.name = name }
    fn speak() { pr "{self.name} makes a sound" }
}

` Single inheritance `
Clas Dog(Animal) {
    def(name: str) {
        sup(name)           ` call parent __init__ `
        self.tricks = []
    }

    fn speak() {
        pr "{self.name} barks!"
    }

    fn learn(trick: str) {
        self.tricks.push(trick)
    }
}

` Multiple inheritance `
Clas Flyable {
    fn fly() { pr "{self.name} is flying" }
}

Clas FlyingDog(Dog, Flyable) {
    def(name: str) { sup(name) }
}

fd = FlyingDog("Astro")
fd.speak()
fd.fly()

Abstract Classes

.ry
abst Clas Shape {
    ` Abstract method — must be implemented by subclasses `
    abst fn area() { }
    abst fn perimeter() { }

    ` Concrete method using abstract methods `
    fn describe() {
        pr "area={self.area()}  perim={self.perimeter()}"
    }
}

Clas Circle(Shape) {
    def(r: flo) { self.r = r }
    fn area()      { ret 3.14159 * self.r * self.r }
    fn perimeter() { ret 2 * 3.14159 * self.r }
}

c = Circle(5)
c.describe()

Properties (Getters & Setters)

.ry
Clas Temperature {
    def(celsius: flo) {
        self._c = celsius
    }

    ` Getter — accessed as  obj.fahrenheit `
    get fn fahrenheit() {
        ret self._c * 9 / 5 + 32
    }

    ` Setter — assigned as  obj.celsius = val `
    get fn celsius() {
        ret self._c
    }

    set fn celsius(val) {
        if val < -273.15 {
            thr "Below absolute zero!"
        }
        self._c = val
    }
}

t = Temperature(100)
pr "Boiling: {t.fahrenheit} F"
t.celsius = 0
pr "Freezing: {t.fahrenheit} F"

Error Handling

.ry
` Basic try/catch `
try {
    result = 10 / 0
} catch {
    pr "Something went wrong"
}

` Typed catch `
try {
    n = int("not a number")
} catch ValueError {
    pr "Bad number format"
}

` Multi-type catch `
try {
    data = fetchData()
} catch ValueError, TypeError {
    pr "Data error"
} catch Exception {
    pr "General error"
} fin {
    pr "Cleanup — always runs"
}

` Throw / raise `
fn divide(a, b) {
    if b == 0 { thr "Division by zero" }
    ret a / b
}

` Null coalescing as safe fallback `
val = riskyFunc() ?? "default"

Generators

Use yld to create a generator function:

.ry
fn counter(limit) {
    i = 0
    while i < limit {
        yld i
        i++
    }
}

for n in counter(5) {
    pr n     ` 0 1 2 3 4 `
}

fn fibonacci() {
    a, b = 0, 1
    while TRUE {
        yld a
        a, b = b, a + b
    }
}

gen = fibonacci()
for i in 0..10 {
    pr next(gen)
}

Imports & Modules

.ry
` Import a Python module `
imp math
pr math.sqrt(144)

` Import specific symbol `
imp sqrt from math
pr sqrt(25)

` Import with alias `
imp numpy as np

` Wildcard import `
imp * from math
pr pi
pr floor(3.9)

` Import a .ry module (auto-compiled) `
imp myUtils
pr myUtils.factorial(10)

.ry Module Imports Phase 7

You can split your Rattled code across multiple .ry files. When you imp myModule and myModule.ry exists in the same directory, Rattled automatically compiles it and makes it available as a module object.

.ry mathUtils.ry
fn square(n) { ret n * n }
fn cube(n)   { ret n * n * n }

fn factorial(n) {
    if n <= 1 { ret 1 }
    ret n * factorial(n - 1)
}
.ry main.ry
imp mathUtils

pr mathUtils.square(5)        ` 25  `
pr mathUtils.cube(3)          ` 27  `
pr mathUtils.factorial(6)     ` 720 `
💡

The .ry module is compiled at import time and injected into sys.modules, so it behaves exactly like a Python module. Circular imports are detected and skipped.
