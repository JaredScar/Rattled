See the syntax

Explore Rattled's concise, expressive language features.

shapes.ry 
abst Clas Shape {
    abst fn area() { }

    get fn label() {
        ret "Shape(area={self.area()})"
    }
}

Clas Circle(Shape) {
    def(radius: flo) {
        self.radius = radius
    }

    fn area() -> flo {
        ret 3.14159 * self.radius * self.radius
    }
}

Clas Rectangle(Shape) {
    def(w: flo, h: flo) {
        self.w = w
        self.h = h
    }

    fn area() -> flo { ret self.w * self.h }
}

shapes = [Circle(5), Rectangle(4, 6)]
for shape in shapes {
    pr shape.label
}

errors.ry 
fn safeDivide(a: flo, b: flo) -> flo {
    if b == 0 {
        thr "Division by zero!"
    }
    ret a / b
}

` Multi-type catch + finally `
try {
    result = safeDivide(10, 0)
    pr result
} catch ValueError, RuntimeError {
    pr "Caught a value or runtime error"
} catch Exception {
    pr "Caught general exception"
} fin {
    pr "Always runs"
}

` Null coalescing `

Clas Counter {
    def(start: int = 0) {
        self.value = start
    }

    fn increment(by: int = 1) {
        self.value += by
    }

    get fn count() {
        ret self.value
    }
}

c = Counter(10)
c.increment(5)
pr "count = {c.count}"
🐍 Python (equivalent)
class Counter:
    def __init__(self, start: int = 0):
        self.value = start

    def increment(self, by: int = 1):
        self.value += by

    @property
    def count(self):
        return self.value

c = Counter(10)
c.increment(5)
print(f"count = {c.count}")
